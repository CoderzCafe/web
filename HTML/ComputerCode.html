            public class Person { 
                private String name; 
                private String age; 

                public Person(String name, String age) {
                    this.name = name;
                    this.age = age;
                }

                public void setName(String name) {
                    this.name = name;
                }
                public String getName() {
                    return name;
                }

                public static void main(String[] args) {
                    System.out.println("Name " +new Person("Shahin", "24").getName());                    
                }
            }

Output is
name Shahin
